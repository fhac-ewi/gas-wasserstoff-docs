\newcommand{\image}[2]{
  \begin{figure}[!htp]
    \centering
    \includegraphics[max height=19cm,max width=0.9\textwidth,keepaspectratio]{images/#1}
    \caption{#2}
    \label{image:#1}
  \end{figure}
  \FloatBarrier
} % Beispiel: \image{Funny.PNG}{Dies ist eine Beschreibung}
